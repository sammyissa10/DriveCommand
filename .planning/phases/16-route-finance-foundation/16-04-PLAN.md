**File 2: src/lib/validations/expense-template.schemas.ts**

```typescript
import { z } from 'zod';

const templateItemSchema = z.object({
  categoryId: z.string().uuid('Invalid category'),
  amount: z.coerce
    .number()
    .positive('Amount must be positive')
    .multipleOf(0.01)
    .max(999999.99),
  description: z.string().min(1).max(200),
});

export const templateCreateSchema = z.object({
  name: z.string().min(1, 'Template name is required').max(100),
  items: z.array(templateItemSchema).min(1, 'Template must have at least one item'),
});

export const templateUpdateSchema = templateCreateSchema.partial();

export type TemplateCreate = z.infer<typeof templateCreateSchema>;
export type TemplateUpdate = z.infer<typeof templateUpdateSchema>;
```

   - Fetch template with items.
   - Use a Prisma transaction to create RouteExpense records from all template items:
     ```typescript
     await prisma.$transaction(async (tx) => {
       const template = await tx.expenseTemplate.findUnique({
         where: { id: templateId },
         include: { items: true }
       });
       if (!template) throw new Error('Template not found');

       await tx.routeExpense.createMany({
         data: template.items.map(item => ({
           tenantId,
           routeId,
           categoryId: item.categoryId,
           amount: item.amount,
           description: item.description,
         }))
       });
     });
     ```
   - Revalidate `/routes/${routeId}`.
   - Return `{ success: true, count: template.items.length }`.
